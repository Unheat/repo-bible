Authentication Flow

The auth.ts module brokers OAuth handshakes between the client and GitHub's identity service. It is invoked by ingest.ts before any tree fetch and persists the token on the repositories row.

// generated · deep-dive agent
export async function authorize(repo: Repository) {
  const token = await github.exchange(repo);
  return db.update(repo.id, { token });
}
